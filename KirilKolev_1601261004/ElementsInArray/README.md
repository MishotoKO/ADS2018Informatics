<?php
function getMax( $array )
{
   $n 	=	count( $array );
   $max =	$array[0];
   $arr =	[];
   for ( $i = 1; $i < $n; $i++ )
   {
       if ( $max < $array[$i] )
       {
           $max =	$array[$i];
       }
   }

   for ( $i = 1; $i < $n; $i++ )
   {
       if ( $max == $array[$i] )
       {
           $a =	$i;
       }
   }

    array_push( $arr, $max, $a );

    return $arr;
}

function getMin( $array )
{ 
   $n 	=	count($array);
   $min =	$array[0];
   for ( $i = 1; $i < $n; $i++ )
   {
       if ( $min > $array[$i] )
       {
           $min =	$array[$i];
       }
    }
    return $min;
}

function numpat( $n ) 
{
    $num =	$n[0];

    for ( $i = 0; $i < count( $n ); $i++ )
    { 
        for ( $j = 0; $j <= $i; $j++ )
        {
            $num =	$n[$j];
            echo $num . " ";
        } 
        echo "\n";
    } 
} 

$array = array( 56, 12, 89, 65, 61, 36, 45, 23, 55 );
